_renderIsMfcDisbursementUpdated() {
  // TODO: warunek widocznoĹ›ci â€“ odnowienie UW z KI pobranym z ICBS.
  if (!this.shouldRender('isMfcDisbursementUpdated')) return nothing;

  return html`
    <div class="row">
      <div class="col-4 form-label">
        <p>${this.localize('ProductParameters.isMfcDisbursementUpdated')}</p>
      </div>
      <div class="col-6">
        <div class="row">
          <div class="col-12">
            <ing-radio-group
              at-id="isMfcDisbursementUpdated"
              name="isMfcDisbursementUpdated"
              .disabled=${this.isDisabled({ sectionName: 'amountSection', propName: 'isMfcDisbursementUpdated' })}>
              <div
                class="radio-group"
                @change=${e => this.updateProp('isMfcDisbursementUpdated', e.target.value)}>
                <ing-radio
                  at-id="yes"
                  name="isMfcDisbursementUpdated[]"
                  ?checked=${this.productData.isMfcDisbursementUpdated === 'yes'}
                  .choiceValue=${'yes'}
                  label=${this._getDescription({ fieldName: 'yes' })}
                  .disabled=${this.isDisabled({ sectionName: 'amountSection', propName: 'isMfcDisbursementUpdated' })}>
                </ing-radio>
                <ing-radio
                  at-id="no"
                  name="isMfcDisbursementUpdated[]"
                  ?checked=${this.productData.isMfcDisbursementUpdated === 'no'}
                  .choiceValue=${'no'}
                  label=${this._getDescription({ fieldName: 'no' })}
                  .disabled=${this.isDisabled({ sectionName: 'amountSection', propName: 'isMfcDisbursementUpdated' })}>
                </ing-radio>
              </div>
            </ing-radio-group>

            <div class="line">
              <div class="content">
                <cms2-info-message
                  at-id="isMfcDisbursementUpdatedTooltip"
                  .message=${this.productParametersTooltips.isMfcDisbursementUpdatedTooltip}
                  font16>
                </cms2-info-message>
              </div>
            </div>
          </div>
        </div>
      </div>
    </div>

    ${this._renderMfcDisbursementDates()}
  `;
}

_renderMfcDisbursementDates() {
  if (this.productData.isMfcDisbursementUpdated !== 'no') return nothing;

  const today = new Date();

  return html`
    <div class="row">
      <div class="col-4 form-label">
        <p>${this.localize('ProductParameters.mfcDisbursementDatesLabel')}</p>
      </div>
      <div class="col-8">
        <div class="row">
          <div class="col-6">
            <div class="input-container">
              <cms2-date-input
                at-id="mfcDisbursementDateFrom"
                id="mfcDisbursementDateFrom"
                label=${this.localize('ProductParameters.dateFrom')}
                .modelValue=${this.productData.mfcDisbursementDateFrom}
                modelKey="mfcDisbursementDateFrom"
                .disabled=${this.isDisabled({ sectionName: 'amountSection', propName: 'mfcDisbursementDateFrom' })}
                .errorValidators=${[
                  requiredField(this.productData.productType, { fieldName: 'mfcDisbursementDateFrom' }),
                  new MaxDate(today, { getMessage: () => this.localize('ProductParameters.dateMustBeInThePast') }),
                ]}
                @model-value-changed=${this._updatePropHandle}>
              </cms2-date-input>
            </div>
          </div>
          <div class="col-6">
            <div class="input-container">
              <cms2-date-input
                at-id="mfcDisbursementDateTo"
                id="mfcDisbursementDateTo"
                label=${this.localize('ProductParameters.dateTo')}
                .modelValue=${this.productData.mfcDisbursementDateTo}
                modelKey="mfcDisbursementDateTo"
                .disabled=${this.isDisabled({ sectionName: 'amountSection', propName: 'mfcDisbursementDateTo' })}
                .errorValidators=${[
                  requiredField(this.productData.productType, { fieldName: 'mfcDisbursementDateTo' }),
                  new MaxDate(today, { getMessage: () => this.localize('ProductParameters.dateMustBeInThePast') }),
                ]}
                @model-value-changed=${this._updatePropHandle}>
              </cms2-date-input>
            </div>
          </div>
        </div>
      </div>
    </div>
  `;
}
