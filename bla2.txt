    @Test
    public void shouldFillCollateralRenewalFieldsInCollateralProviders() {
        Provider collateralProvider = new Provider();
        collateralProvider.setCardIndex("123");

        CivilGuaranty civilGuaranty = new CivilGuaranty();
        civilGuaranty.setChangingCollateral(YES);
        civilGuaranty.setAnnexChange(Collections.emptyList());
        civilGuaranty.setChangeType("Grande type");
        civilGuaranty.setCollateralAnnexNumber(777);
        civilGuaranty.setCollateralAgreementNumber("Some agreement number");
        civilGuaranty.setCollateralAgreementSignatureDate(
                DateTimeFormatterUtil.parseCmsDate("01-01-2024"));
        civilGuaranty.setProviders(Collections.singleton(collateralProvider));

        CollateralFormDataDTO collateral = new CollateralFormDataDTO();
        collateral.setData(civilGuaranty);

        CivilGuarantyCollateralPrintDataDTO printDataDTO =
                new CivilGuarantyCollateralPrintDataDTO(collateral, prepareEmptyDocumentData());

        assertTrue(CollectionUtils.isNotEmpty(printDataDTO.collateralProviders));
        assertEquals(1, printDataDTO.collateralProviders.size());

        CollateralAgreementProviderDTO collateralAgreementProviderDTO =
                printDataDTO.collateralProviders.get(0);

        assertEquals(YES, collateralAgreementProviderDTO.getChangingCollateral());
        assertTrue(Objects.nonNull(collateralAgreementProviderDTO.getAnnexChange()));
        assertFalse(collateralAgreementProviderDTO.getAnnexChange().getBankAccount());
        assertFalse(collateralAgreementProviderDTO.getAnnexChange().getCollateralProviderData());
        assertFalse(collateralAgreementProviderDTO.getAnnexChange().getAddDealWithOldDebtor());
        assertFalse(collateralAgreementProviderDTO.getAnnexChange().getRemoveDealWithOldDebtor());
        assertEquals("Grande type", collateralAgreementProviderDTO.getChangeType());
        assertEquals(777, collateralAgreementProviderDTO.getCollateralAnnexNumber());
        assertEquals("Some agreement number",
                collateralAgreementProviderDTO.getOldCollateralAgreementNumber());
        assertEquals("01-01-2024",
                collateralAgreementProviderDTO.getCollateralAgreementSignatureDate());
    }

    // =========================================================================
    // Metody pomocnicze
    // =========================================================================

    private CivilGuaranty createCivilGuaranty() {
        CivilGuaranty civilGuaranty = new CivilGuaranty();
        Provider provider = new Provider();
        provider.setCardIndex("123");
        civilGuaranty.setProviders(Collections.singleton(provider));
        return civilGuaranty;
    }

    private CollateralFormDataDTO createCollateral(CivilGuaranty civilGuaranty) {
        CollateralFormDataDTO collateral = new CollateralFormDataDTO();
        collateral.setCollateralType(CollateralType.CIVIL_GUARANTY.name());
        collateral.setData(civilGuaranty);
        return collateral;
    }

    private CollateralAgreementProviderDTO getFirstProvider(
            CivilGuarantyCollateralPrintDataDTO dto) {
        assertTrue(CollectionUtils.isNotEmpty(dto.getCollateralProviders()));
        return dto.getCollateralProviders().get(0);
    }

    private void assertBigDecimalEquals(BigDecimal expected, BigDecimal actual) {
        assertNotNull(actual, "nominalValue should not be null");
        assertEquals(0, expected.compareTo(actual),
                "Expected " + expected + " but got " + actual);
    }

    private DocumentData prepareEmptyDocumentData() {
        return DocumentData.builder()
                .agreementDetailsModel(new AgreementDetailsModel())
                .productParamsModel(new ProductParamsModel())
                .providers(Collections.emptyMap())
                .build();
    }

    private DocumentData buildDocumentData(String productType,
                                            BigDecimal mcRcfAmount,
                                            String mcRcfRepaymentDate,
                                            String creditFacilityRepaymentDate) {
        ProductParamsModel productParamsModel = new ProductParamsModel();
        productParamsModel.setProductType(productType);
        if (mcRcfAmount != null) {
            productParamsModel.setCreditFacilityAmount(mcRcfAmount);
        }

        AgreementDetailsModel agreementDetailsModel = new AgreementDetailsModel();
        if (mcRcfRepaymentDate != null) {
            agreementDetailsModel.setMcfRcfRepaymentDate(mcRcfRepaymentDate);
        }
        if (creditFacilityRepaymentDate != null) {
            agreementDetailsModel.setCreditFacilityRepaymentDate(creditFacilityRepaymentDate);
        }

        return DocumentData.builder()
                .productParamsModel(productParamsModel)
                .agreementDetailsModel(agreementDetailsModel)
                .providers(Collections.emptyMap())
                .collaterals(Collections.emptyList())
                .build();
    }

    private DocumentData buildDocumentDataWithBgk(String productType,
                                                   BigDecimal mcRcfAmount,
                                                   String mcRcfRepaymentDate,
                                                   String creditFacilityRepaymentDate,
                                                   String bgkGuarantyType) {
        ProductParamsModel productParamsModel = new ProductParamsModel();
        productParamsModel.setProductType(productType);
        if (mcRcfAmount != null) {
            productParamsModel.setCreditFacilityAmount(mcRcfAmount);
        }

        AgreementDetailsModel agreementDetailsModel = new AgreementDetailsModel();
        if (mcRcfRepaymentDate != null) {
            agreementDetailsModel.setMcfRcfRepaymentDate(mcRcfRepaymentDate);
        }
        if (creditFacilityRepaymentDate != null) {
            agreementDetailsModel.setCreditFacilityRepaymentDate(creditFacilityRepaymentDate);
        }

        BgkGuarantee bgkGuarantee = new BgkGuarantee();
        bgkGuarantee.setBgkGuarantyType(bgkGuarantyType);

        CollateralFormDataDTO bgkCollateral = new CollateralFormDataDTO();
        bgkCollateral.setCollateralType(CollateralType.BGK_GUARANTEE.name());
        bgkCollateral.setData(bgkGuarantee);

        return DocumentData.builder()
                .productParamsModel(productParamsModel)
                .agreementDetailsModel(agreementDetailsModel)
                .providers(Collections.emptyMap())
                .collaterals(Collections.singletonList(bgkCollateral))
                .build();
    }
