@Test
public void shouldCalculateCollateralEndDateForCivilGuaranty_whenEnterDateBeforeRepayment() {
    // given
    DisbursementInScenarioContext context = prepareContext();

    CivilGuaranty civilGuaranty = new CivilGuaranty();
    civilGuaranty.setCollateralEndPeriod("enterDate");
    civilGuaranty.setCollateralEndDate(LocalDate.of(2026, 4, 10));

    CollateralFormDataDTO dto = new CollateralFormDataDTO();
    dto.setCollateralType(CollateralType.CIVIL_GUARANTY.name());
    dto.setData(civilGuaranty);
    context.setCollaterals(Collections.singletonList(dto));

    ProductParamsModel productParamsModel = new ProductParamsModel();
    productParamsModel.setProductType("multiproductCreditFacility");
    context.setProductParams(productParamsModel);

    AgreementDetailsModel agreementDetailsModel = new AgreementDetailsModel();
    agreementDetailsModel.setMcfRcfRepaymentDate("10-05-2026");
    context.setAgreementDetailsModel(agreementDetailsModel);

    // when
    CollateralRecordRequest recordRequest = mapper.map(context);

    // then
    CivilGuaranty result = (CivilGuaranty) recordRequest
            .getCollateralFormDataList().get(0).getData();
    assertEquals(LocalDate.of(2029, 4, 10), result.getCollateralEndDate());
}
