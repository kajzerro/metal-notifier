_renderMcfAvailabilityPeriod() {
  return html`
    <h2 class="section-header">${this.localize('ProductParameters.mcfAvailabilityPeriod')}</h2>
    ${this._renderEditableSectionInfo({ sectionName: 'amountSection' })}

    <div class="row">
      <div class="col-4 form-label-flex">
        <p>${this.localize('ProductParameters.mcfDisbursementPeriodLength')}</p>
      </div>
      <div class="col-8">
        <div class="input-container">
          <cms2-dict-select
            at-id="mcfDisbursementPeriodLength"
            id="mcfDisbursementPeriodLength"
            .modelValue=${this.productData.mcfDisbursementPeriodLength}
            disabled
            .dictionary=${this.dictionariesService.getProductCreditFacilityPeriodDictionary(this.productData)}
            modelKey="mcfDisbursementPeriodLength"
            @model-value-changed=${this._updatePropHandle}>
          </cms2-dict-select>
        </div>
      </div>
    </div>

    ${this._renderIsMfcDisbursementUpdated()}
  `;
}

_renderIsMfcDisbursementUpdated() {
  // TODO: warunek widoczności – odnowienie UW z KI pobranym z ICBS.
  // Podmień poniższy guard na właściwy getter, np.:
  // if (!this.isMfcRenewalWithIcbsInvestmentCredit) return nothing;
  if (!this.shouldRender('isMfcDisbursementUpdated')) return nothing;

  return html`
    <div class="row">
      <div class="col-4 form-label">
        <p>${this.localize('ProductParameters.isMfcDisbursementUpdated')}</p>
      </div>
      <div class="col-6">
        <div class="row">
          <div class="col-12">
            <ing-radio-group
              at-id="isMfcDisbursementUpdated"
              name="isMfcDisbursementUpdated"
              .disabled=${this.isDisabled({ sectionName: 'amountSection', propName: 'isMfcDisbursementUpdated' })}>
              <div
                class="radio-group"
                @change=${e => this.updateProp('isMfcDisbursementUpdated', e.target.value)}>
                <ing-radio
                  at-id="yes"
                  name="isMfcDisbursementUpdated[]"
                  ?checked=${this.productData.isMfcDisbursementUpdated === 'yes'}
                  .choiceValue=${'yes'}
                  label=${this._getDescription({ fieldName: 'yes' })}
                  .disabled=${this.isDisabled({ sectionName: 'amountSection', propName: 'isMfcDisbursementUpdated' })}>
                </ing-radio>
                <ing-radio
                  at-id="no"
                  name="isMfcDisbursementUpdated[]"
                  ?checked=${this.productData.isMfcDisbursementUpdated === 'no'}
                  .choiceValue=${'no'}
                  label=${this._getDescription({ fieldName: 'no' })}
                  .disabled=${this.isDisabled({ sectionName: 'amountSection', propName: 'isMfcDisbursementUpdated' })}>
                </ing-radio>
              </div>
            </ing-radio-group>

            <div class="line">
              <div class="content">
                <cms2-info-message
                  at-id="isMfcDisbursementUpdatedTooltip"
                  .message=${this.productParametersTooltips.isMfcDisbursementUpdatedTooltip}
                  font16>
                </cms2-info-message>
              </div>
            </div>
          </div>
        </div>
      </div>
    </div>
  `;
}
